       ```sql
       CREATE EXTENSION IF NOT EXISTS btree_gist;

       ALTER TABLE reservations
       ADD CONSTRAINT no_overlapping_reservations
       EXCLUDE USING GIST (
           space_id WITH =,
           tstzrange(start_time, end_time, '[)') WITH &&
       )
       WHERE (status IN ('pending', 'confirmed'));
       ```
