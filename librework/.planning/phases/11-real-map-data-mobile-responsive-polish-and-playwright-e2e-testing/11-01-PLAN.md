  2. Add debounce ref: `const moveDebounceRef = useRef<ReturnType<typeof setTimeout> | null>(null)`.

  3. Add `handleBoundsChange` callback using `useCallback`:
     ```
     const handleBoundsChange = useCallback((center: { lat: number; lng: number }) => {
       if (moveDebounceRef.current) clearTimeout(moveDebounceRef.current)
       moveDebounceRef.current = setTimeout(() => {
         if (!process.env.NEXT_PUBLIC_GOOGLE_MAPS_API_KEY) return
         setIsLoading(true)
         fetchNearbyEstablishments(center, 2000, userCenter ?? undefined)
           .then(results => { if (results.length > 0) setLiveEstablishments(results) })
           .finally(() => setIsLoading(false))
       }, 500)
     }, [userCenter])
     ```

  4. Pass `onBoundsChange={handleBoundsChange}` to the MapView component in the JSX.

  5. Update the existing `useEffect` that calls `fetchNearbyEstablishments` on mount to also pass `userCenter` for distance calculation: `fetchNearbyEstablishments(center, 2000, userCenter ?? undefined)`.
