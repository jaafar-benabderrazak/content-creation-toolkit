```python
"""
Tests for config.PipelineConfig schema validation (CONF-04).

Run with: pytest tests/test_pipeline_config.py -v
"""
from __future__ import annotations

import tempfile
from pathlib import Path

import pytest
import yaml

from config import PipelineConfig
from config.pipeline_config import VideoSettings, PublishSettings, NotifySettings, PostSettings


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _write_yaml(data: dict, tmp_path: Path) -> Path:
    p = tmp_path / "config.yaml"
    p.write_text(yaml.dump(data, default_flow_style=False), encoding="utf-8")
    return p


VALID_MINIMAL = {
    "profile": "test-profile",
    "video": {
        "style_prompt": "test style",
        "mood": "test",
        "duration_minutes": 5,
    }
}


# ---------------------------------------------------------------------------
# Construction tests
# ---------------------------------------------------------------------------

def test_default_construction():
    """PipelineConfig() must construct with all defaults — no arguments required."""
    cfg = PipelineConfig()
    assert cfg.profile == "default"
    assert cfg.video.duration_minutes == 120
    assert cfg.publish.youtube_enabled is False
    assert cfg.notify.require_approval is True


def test_sub_model_defaults():
    """All nested sub-models must have correct defaults."""
    cfg = PipelineConfig()
    assert cfg.video.resolution == "1080p"
    assert cfg.video.quality_preset == "high"
    assert cfg.post.watermark_enabled is False
    assert cfg.publish.youtube_privacy == "private"
    assert cfg.notify.approval_timeout_seconds == 3600


# ---------------------------------------------------------------------------
# from_yaml — valid input
# ---------------------------------------------------------------------------

def test_from_yaml_valid_minimal(tmp_path):
    """A minimal valid YAML with only a subset of fields loads without error."""
    p = _write_yaml(VALID_MINIMAL, tmp_path)
    cfg = PipelineConfig.from_yaml(p)
    assert cfg.profile == "test-profile"
    assert cfg.video.mood == "test"
    # Unset fields fall back to defaults
    assert cfg.video.enable_parallax is True


def test_from_yaml_named_profile(tmp_path):
    """Each bundled named profile loads without ValidationError."""
    profiles_dir = Path("config/profiles")
    for yaml_file in profiles_dir.glob("*.yaml"):
        cfg = PipelineConfig.from_yaml(yaml_file)
        assert cfg.profile != "default", f"{yaml_file.name} did not set a profile name"
        assert len(cfg.video.style_prompt) > 10, f"{yaml_file.name} style_prompt too short"


# ---------------------------------------------------------------------------
# from_yaml — invalid input / ValidationError paths
# ---------------------------------------------------------------------------

def test_from_yaml_wrong_type_duration(tmp_path):
    """duration_minutes must be an integer — string value must raise ValidationError."""
    data = {**VALID_MINIMAL, "video": {**VALID_MINIMAL["video"], "duration_minutes": "two-hours"}}
    p = _write_yaml(data, tmp_path)
    with pytest.raises(Exception) as exc_info:
        PipelineConfig.from_yaml(p)
    # Must mention the offending field somewhere in the error message
    err = str(exc_info.value).lower()
    assert "duration_minutes" in err or "duration" in err, (
        f"Error should mention 'duration_minutes', got: {exc_info.value}"
    )


def test_from_yaml_invalid_resolution(tmp_path):
    """resolution must match pattern '^(1080p|720p|480p)$' — '4K' must fail."""
    data = {**VALID_MINIMAL, "video": {**VALID_MINIMAL["video"], "resolution": "4K"}}
    p = _write_yaml(data, tmp_path)
    with pytest.raises(Exception) as exc_info:
        PipelineConfig.from_yaml(p)
    err = str(exc_info.value).lower()
    assert "resolution" in err or "pattern" in err or "4k" in err, (
        f"Error should mention 'resolution', got: {exc_info.value}"
    )


def test_from_yaml_invalid_youtube_privacy(tmp_path):
    """youtube_privacy must be 'public', 'unlisted', or 'private' — 'secret' must fail."""
    data = {
        **VALID_MINIMAL,
        "publish": {"youtube_privacy": "secret"}
    }
    p = _write_yaml(data, tmp_path)
    with pytest.raises(Exception) as exc_info:
        PipelineConfig.from_yaml(p)
    err = str(exc_info.value).lower()
    assert "privacy" in err or "pattern" in err or "secret" in err, (
        f"Error should mention 'privacy', got: {exc_info.value}"
    )


def test_from_yaml_duration_below_minimum(tmp_path):
    """duration_minutes ge=1 — value of 0 must fail."""
    data = {**VALID_MINIMAL, "video": {**VALID_MINIMAL["video"], "duration_minutes": 0}}
    p = _write_yaml(data, tmp_path)
    with pytest.raises(Exception) as exc_info:
        PipelineConfig.from_yaml(p)
    err = str(exc_info.value).lower()
    assert "duration_minutes" in err or "greater" in err or "ge" in err, (
        f"Error should mention 'duration_minutes' constraint, got: {exc_info.value}"
    )


def test_from_yaml_empty_file(tmp_path):
    """An empty YAML file must raise ValueError, not silently return None."""
    p = tmp_path / "empty.yaml"
    p.write_text("", encoding="utf-8")
    with pytest.raises((ValueError, Exception)) as exc_info:
        PipelineConfig.from_yaml(p)
    assert "empty" in str(exc_info.value).lower() or exc_info.type is not None


# ---------------------------------------------------------------------------
# File not found
# ---------------------------------------------------------------------------

def test_from_yaml_file_not_found():
    """from_yaml on a nonexistent path must raise FileNotFoundError."""
    with pytest.raises(FileNotFoundError):
        PipelineConfig.from_yaml(Path("nonexistent_config_file_xyz.yaml"))


# ---------------------------------------------------------------------------
# YAML round-trip
# ---------------------------------------------------------------------------

def test_yaml_round_trip(tmp_path):
    """from_yaml -> to_yaml -> from_yaml must produce identical field values."""
    original = PipelineConfig.from_yaml(Path("config/profiles/lofi_study.yaml"))
    saved_path = tmp_path / "round_trip.yaml"
    original.to_yaml(saved_path)
    reloaded = PipelineConfig.from_yaml(saved_path)

    assert reloaded.profile == original.profile
    assert reloaded.video.style_prompt == original.video.style_prompt
    assert reloaded.video.mood == original.video.mood
    assert reloaded.video.duration_minutes == original.video.duration_minutes
    assert reloaded.publish.youtube_enabled == original.publish.youtube_enabled
    assert reloaded.notify.require_approval == original.notify.require_approval


def test_to_yaml_creates_valid_file(tmp_path):
    """to_yaml must write a YAML file that can be parsed by yaml.safe_load."""
    cfg = PipelineConfig()
    out_path = tmp_path / "out.yaml"
    cfg.to_yaml(out_path)
    assert out_path.exists()
    data = yaml.safe_load(out_path.read_text(encoding="utf-8"))
    assert isinstance(data, dict)
    assert "profile" in data
    assert "video" in data
```
