- Aggregate via `sklearn.cluster.KMeans(n_clusters=5, n_init=10, random_state=42)` on collected colors
- Convert cluster centers to `[int, int, int]` lists
- Compute brightness/contrast from PIL: `img.convert("L")` â†’ `np.array()` â†’ `.mean()` / `.std()`
- Color temperature: avg_r - avg_b > 15 â†’ "warm"; avg_b - avg_r > 15 â†’ "cool"; else "neutral"
- Brightness thresholds: < 85 â†’ "low", > 170 â†’ "high", else "medium"
- Contrast thresholds: std < 35 â†’ "low", std > 70 â†’ "high", else "medium"
- Derive `mood_descriptors` list from feature combination:
  - warm + low brightness â†’ ["moody", "warm", "atmospheric", "cinematic"]
  - cool + high contrast â†’ ["sharp", "cool", "dramatic", "editorial"]
  - neutral + medium â†’ ["balanced", "clean", "natural"]

**StyleReferenceManager class**:
```python
class StyleReferenceManager:
    def __init__(self, cache_dir: Path = Path(".cache/style_reference")):
        self.cache_dir = cache_dir

    def profile_path(self, handle: str) -> Path:
        return self.cache_dir / handle.lstrip("@") / "profile.json"

    def posts_dir(self, handle: str) -> Path:
        return self.cache_dir / handle.lstrip("@") / "posts"

    def load(self, handle: str) -> Optional[StyleProfile]:
        """Load existing profile.json without re-scraping."""
        p = self.profile_path(handle)
        if not p.exists():
            return None
        data = json.loads(p.read_text(encoding="utf-8"))
        return StyleProfile(**data)

    def build_or_load(
        self,
        handle: str,
        session_file: Optional[str] = None,
        post_limit: int = 30,
        refresh: bool = False,
    ) -> StyleProfile:
        """Load cached profile or scrape + extract if absent or refresh=True."""
        if not refresh:
            cached = self.load(handle)
            if cached:
                logger.info(f"[StyleRef] Loaded cached profile for @{handle}")
                return cached

        posts_dir = self.posts_dir(handle)
        existing_images = list(posts_dir.glob("*.jpg")) if posts_dir.exists() else []

        if existing_images and not refresh:
            # Manual fallback path: user-provided images, no scraping
            logger.info(f"[StyleRef] Using {len(existing_images)} manually placed images for @{handle}")
            image_paths = existing_images
        else:
            # Scrape
            image_paths = scrape_profile_images(handle, post_limit, self.cache_dir, session_file)

        if not image_paths:
            raise RuntimeError(f"No images found for @{handle}. Place .jpg files in {posts_dir}")

        return build_style_profile(handle, image_paths, self.cache_dir)

    def get_reference_image_paths(self, handle: str, max_refs: int = 10) -> list[Path]:
        """Return reference image paths for this handle (from cached profile)."""
        profile = self.load(handle)
        if not profile:
            raise RuntimeError(f"No style profile for @{handle}. Run build_or_load() first.")
        return [Path(p) for p in profile.reference_images[:max_refs] if Path(p).exists()]
```
