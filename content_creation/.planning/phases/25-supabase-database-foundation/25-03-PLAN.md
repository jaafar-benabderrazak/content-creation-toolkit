    **dashboard/src/app/api/roadmap/route.ts** — replaces the pipelineGet/pipelinePost proxy:
    ```typescript
    export const runtime = "nodejs";
    import { supabase } from "@/lib/supabase";

    export async function GET(req: Request) {
      const url = new URL(req.url);
      const status = url.searchParams.get("status") || "";

      let query = supabase
        .from("videos")
        .select("*")
        .order("position", { ascending: true });

      if (status) {
        query = query.eq("status", status);
      }

      const { data, error } = await query;
      if (error) {
        return Response.json({ error: error.message, entries: [] }, { status: 500 });
      }
      return Response.json({ entries: data ?? [] });
    }

    export async function POST(req: Request) {
      const body = await req.json();

      // Determine next position
      const { data: last } = await supabase
        .from("videos")
        .select("position")
        .order("position", { ascending: false })
        .limit(1);

      const nextPosition = last && last.length > 0 ? last[0].position + 1 : 0;

      const now = new Date().toISOString();
      const row = {
        id: body.id ?? crypto.randomUUID().replace(/-/g, ""),
        title: body.title ?? "",
        tags: body.tags ?? "",
        profile: body.profile ?? "lofi-study",
        notes: body.notes ?? "",
        status: body.status ?? "planned",
        position: nextPosition,
        created_at: body.created_at ?? now,
        updated_at: now,
      };

      const { data, error } = await supabase
        .from("videos")
        .insert(row)
        .select()
        .single();

      if (error) {
        return Response.json({ error: error.message }, { status: 500 });
      }
      return Response.json(data, { status: 201 });
    }
    ```
