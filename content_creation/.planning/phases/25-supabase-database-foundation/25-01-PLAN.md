<task type="auto">
  <name>Task 2: Apply 4-table schema via MCP migration</name>
  <files>supabase/migrations/001_initial_schema.sql</files>
  <action>
    Write `supabase/migrations/001_initial_schema.sql` with the following DDL, then apply it using `mcp__plugin_supabase_supabase__apply_migration` with migration_name "initial_schema":

    ```sql
    -- videos: replaces video_roadmap.json
    -- position column preserves list ordering (move_up / move_down behavior)
    CREATE TABLE IF NOT EXISTS videos (
      id          TEXT PRIMARY KEY,
      title       TEXT NOT NULL,
      tags        TEXT NOT NULL DEFAULT '',
      profile     TEXT NOT NULL DEFAULT 'lofi-study',
      notes       TEXT NOT NULL DEFAULT '',
      status      TEXT NOT NULL DEFAULT 'planned'
                  CHECK (status IN ('planned', 'producing', 'published')),
      position    INTEGER NOT NULL DEFAULT 0,
      created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
      updated_at  TIMESTAMPTZ NOT NULL DEFAULT now()
    );

    CREATE INDEX IF NOT EXISTS videos_status_idx ON videos (status);
    CREATE INDEX IF NOT EXISTS videos_position_idx ON videos (position);

    -- executions: replaces execution_log.json
    CREATE TABLE IF NOT EXISTS executions (
      id          TEXT PRIMARY KEY,
      started_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
      finished_at TIMESTAMPTZ,
      pipeline    TEXT NOT NULL DEFAULT '',
      profile     TEXT NOT NULL DEFAULT '',
      tags        TEXT NOT NULL DEFAULT '',
      output_path TEXT NOT NULL DEFAULT '',
      status      TEXT NOT NULL DEFAULT 'running'
                  CHECK (status IN ('running', 'done', 'failed')),
      exit_code   INTEGER,
      last_lines  TEXT NOT NULL DEFAULT ''
    );

    CREATE INDEX IF NOT EXISTS executions_started_at_idx ON executions (started_at DESC);

    -- prompts: stores all 8 prompt chain sections per video
    -- section values: scene_prompts, style_prompt, music_prompt,
    --   thumbnail_text, youtube_title, youtube_description, youtube_tags, suno_prompt
    CREATE TABLE IF NOT EXISTS prompts (
      id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      video_id    TEXT NOT NULL REFERENCES videos(id) ON DELETE CASCADE,
      section     TEXT NOT NULL,
      content     TEXT NOT NULL DEFAULT '',
      model       TEXT NOT NULL DEFAULT '',
      created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
      UNIQUE (video_id, section)
    );

    CREATE INDEX IF NOT EXISTS prompts_video_id_idx ON prompts (video_id);

    -- settings: key/value store for API keys, webhook URLs, profile configs
    CREATE TABLE IF NOT EXISTS settings (
      key         TEXT PRIMARY KEY,
      value       TEXT NOT NULL DEFAULT '',
      updated_at  TIMESTAMPTZ NOT NULL DEFAULT now()
    );

    -- RLS: enable but allow all for anon (single-user app, no auth yet)
    ALTER TABLE videos ENABLE ROW LEVEL SECURITY;
    ALTER TABLE executions ENABLE ROW LEVEL SECURITY;
    ALTER TABLE prompts ENABLE ROW LEVEL SECURITY;
    ALTER TABLE settings ENABLE ROW LEVEL SECURITY;

    CREATE POLICY "allow_all_videos"     ON videos     FOR ALL TO anon USING (true) WITH CHECK (true);
    CREATE POLICY "allow_all_executions" ON executions FOR ALL TO anon USING (true) WITH CHECK (true);
    CREATE POLICY "allow_all_prompts"    ON prompts    FOR ALL TO anon USING (true) WITH CHECK (true);
    CREATE POLICY "allow_all_settings"   ON settings   FOR ALL TO anon USING (true) WITH CHECK (true);
    ```

    After applying, verify with `mcp__plugin_supabase_supabase__execute_sql`:
    ```sql
    SELECT table_name FROM information_schema.tables
    WHERE table_schema = 'public'
    ORDER BY table_name;
    ```
    Expected result: 4 rows — executions, prompts, settings, videos.
